export const defaultEntityLink = '[User Settings](https://docs.wandb.ai/guides/models/app/settings-page/user-settings/#default-team)';
export const defaultEntityLinkText = 'To find or update your default entity, refer to';

export function DefaultEntityNote({ variant = 'full' }) {
  if (variant === 'inline') {
    return <>If not specified, your default entity is used. {defaultEntityLinkText} {defaultEntityLink} in the W&B Models documentation.</>;
  }
  
  // Default 'full' variant
  return <>If you don't specify a W&B team when you call `weave.init()`, your default entity is used. {defaultEntityLinkText} {defaultEntityLink} in the W&B Models documentation.</>;
}

export default DefaultEntityNote;
